# Axum Skeleton + Auth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create `crates/server` — a standalone Axum binary with email+password auth, TOTP 2FA, session management, admin middleware, server config, and REST API placeholders. JSON-only endpoints (no HTML), testable via curl.

**Architecture:** Axum 0.8 binary imports `crates/db` for all data access. Auth via axum-login 0.18 (`AuthnBackend` with argon2 password verification). Sessions in SQLite via tower-sessions + tower-sessions-sqlx-store. 2FA: password verify stores `pending_user_id` in raw session, TOTP verify completes login via `auth_session.login()`. `Require` middleware protects app routes. Separate `admin_required` middleware for server config endpoints.

**Tech Stack:** axum 0.8, axum-login 0.18, tower-sessions 0.15, tower-sessions-sqlx-store 0.15 (sqlite), argon2 0.6, totp-rs 5.7, tower-http (TraceLayer, CorsLayer), tracing + tracing-subscriber

**Depends on:** Plan 1 (crates/db) must be complete — this plan imports `kartoteka-db`.

**Key design decisions:**
- 2FA flow: `auth_session.login()` only fires AFTER full 2FA. Pre-2FA state stored as `pending_user_id` in raw session. `Require` middleware works unchanged.
- `return_to` in session for post-login redirect (not URL query param). Survives refresh, no open redirect risk.
- First registered user gets `role = 'admin'`. Registration gated by `server_config.registration_enabled`.
- `admin_required` middleware as separate layer (not inline check) for server config routes.
- JSON-only responses. No HTML forms — frontend comes in Plan 3.

```rust
pub struct User {
    pub id: String,
    pub email: String,
    pub name: Option<String>,
    pub avatar_url: Option<String>,
    pub role: String,
    pub created_at: String,
    pub updated_at: String,

#[tracing::instrument(skip(pool))]
pub async fn create(
    pool: &SqlitePool,
    email: &str,
    name: Option<&str>,
    role: &str,
) -> crate::Result<User> {
    let id = uuid::Uuid::new_v4().to_string();
    sqlx::query(
        "INSERT INTO users (id, email, name, role) VALUES (?, ?, ?, ?)"
    )
    .bind(&id).bind(email).bind(name).bind(role)
    .execute(pool).await?;

    find_by_id(pool, &id).await?.ok_or_else(|| crate::DbError::NotFound("user".into()))
}

        "SELECT id, email, name, avatar_url, role, created_at, updated_at FROM users WHERE email = ?"
    )
    .bind(email)
    .fetch_optional(pool).await?;
    Ok(user)
```

```rust
// crates/server/src/error.rs
use axum::http::StatusCode;
use axum::response::{IntoResponse, Response};
use axum::Json;

#[derive(Debug)]
pub enum AppError {
    Unauthorized,
    Forbidden,
    NotFound(String),
    Validation(String),
    Internal(String),
    Db(kartoteka_db::DbError),
}

impl From<kartoteka_db::DbError> for AppError {
    fn from(e: kartoteka_db::DbError) -> Self {
        match e {
            kartoteka_db::DbError::NotFound(msg) => AppError::NotFound(msg),
            kartoteka_db::DbError::Validation(msg) => AppError::Validation(msg),
            other => AppError::Internal(other.to_string()),
        }
    }
}

impl IntoResponse for AppError {
    fn into_response(self) -> Response {
        let (status, code) = match &self {
            AppError::Unauthorized => (StatusCode::UNAUTHORIZED, "unauthorized"),
            AppError::Forbidden => (StatusCode::FORBIDDEN, "forbidden"),
            AppError::NotFound(msg) => (StatusCode::NOT_FOUND, msg.as_str()),
            AppError::Validation(msg) => (StatusCode::UNPROCESSABLE_ENTITY, msg.as_str()),
            AppError::Internal(_) => (StatusCode::INTERNAL_SERVER_ERROR, "internal_error"),
        };
        let body = serde_json::json!({ "error": code });
        (status, Json(body)).into_response()
    }
}
```

```rust
// crates/server/src/main.rs
use std::net::SocketAddr;

use axum::Router;
use axum_login::AuthManagerLayerBuilder;
use sqlx::SqlitePool;
use tower_http::trace::TraceLayer;
use tower_sessions::Expiry;
use tower_sessions_sqlx_store::SqliteStore;
use time::Duration;

mod api;
mod auth;
mod error;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    tracing_subscriber::fmt()
        .with_env_filter(
            tracing_subscriber::EnvFilter::try_from_default_env()
                .unwrap_or_else(|_| "kartoteka_server=debug,tower_http=debug".into()),
        )
        .init();

    let db_url = std::env::var("DATABASE_URL")
        .unwrap_or_else(|_| "sqlite://data.db?mode=rwc".into());
    let pool = SqlitePool::connect(&db_url).await?;
    sqlx::migrate!("../db/migrations").run(&pool).await?;

    // Session store
    let session_store = SqliteStore::new(pool.clone());
    session_store.migrate().await?;

    let deletion_task = tokio::task::spawn(
        session_store.clone().continuously_delete_expired(tokio::time::Duration::from_secs(3600)),
    );

    // Auth layer
    let backend = auth::backend::Backend::new(pool.clone());
    let session_layer = tower_sessions::SessionManagerLayer::new(session_store)
        .with_secure(true)
        .with_http_only(true)
        .with_same_site(tower_sessions::cookie::SameSite::Lax)
        .with_expiry(Expiry::OnInactivity(Duration::days(7)));
    let auth_layer = AuthManagerLayerBuilder::new(backend, session_layer).build();

    let app = Router::new()
        .nest("/api", api::routes(pool.clone()))
        .nest("/auth", auth::handlers::routes(pool.clone()))
        .layer(auth_layer)
        .layer(TraceLayer::new_for_http());

    let port: u16 = std::env::var("PORT")
        .ok()
        .and_then(|p| p.parse().ok())
        .unwrap_or(3000);
    let addr = SocketAddr::from(([0, 0, 0, 0], port));

    tracing::info!("listening on {}", addr);
    let listener = tokio::net::TcpListener::bind(addr).await?;
    axum::serve(listener, app).await?;

    deletion_task.abort();
    Ok(())
}
```

```rust
// crates/server/src/auth/backend.rs
use axum_login::{AuthUser, AuthnBackend};
use kartoteka_db::users::User;
use password_hash::PasswordHash;
use sqlx::SqlitePool;

impl AuthUser for User {
    type Id = String;

    fn id(&self) -> Self::Id {
        self.id.clone()
    }

    fn session_auth_hash(&self) -> &[u8] {
        // Use user id as session hash — password changes don't invalidate
        // sessions (acceptable for personal app). For stricter behavior,
        // store password hash in User and return that here.
        self.id.as_bytes()
    }
}

#[derive(Debug, Clone)]
pub struct Backend {
    pool: SqlitePool,
}

impl Backend {
    pub fn new(pool: SqlitePool) -> Self {
        Self { pool }
    }
}

#[derive(Debug, Clone, serde::Deserialize)]
pub struct Credentials {
    pub email: String,
    pub password: String,
}

impl AuthnBackend for Backend {
    type User = User;
    type Credentials = Credentials;
    type Error = BackendError;

    async fn authenticate(
        &self,
        creds: Self::Credentials,
    ) -> Result<Option<Self::User>, Self::Error> {
        let user = kartoteka_db::users::find_by_email(&self.pool, &creds.email)
            .await
            .map_err(|e| BackendError(e.to_string()))?;

        let Some(user) = user else {
            return Ok(None);
        };

        let method = kartoteka_db::auth_methods::find_by_user_and_provider(
            &self.pool, &user.id, "password",
        )
        .await
        .map_err(|e| BackendError(e.to_string()))?;

        let Some(method) = method else {
            return Ok(None);
        };

        let Some(ref hash_str) = method.credential else {
            return Ok(None);
        };

        let parsed_hash = PasswordHash::new(hash_str)
            .map_err(|e| BackendError(e.to_string()))?;

        let is_valid = argon2::Argon2::default()
            .verify_password(creds.password.as_bytes(), &parsed_hash)
            .is_ok();

        if is_valid {
            Ok(Some(user))
        } else {
            Ok(None)
        }
    }

    async fn get_user(
        &self,
        user_id: &String,
    ) -> Result<Option<Self::User>, Self::Error> {
        kartoteka_db::users::find_by_id(&self.pool, user_id)
            .await
            .map_err(|e| BackendError(e.to_string()))
    }
}

#[derive(Debug, Clone, thiserror::Error)]
#[error("{0}")]
pub struct BackendError(pub String);

pub type AuthSession = axum_login::AuthSession<Backend>;
```

Note: add `password-hash = "0.6"` to `argon2` usage — `argon2` re-exports `password_hash::PasswordHash` but we need the `PasswordHash::new` parser from `password-hash` crate. Actually `argon2` 0.6 depends on `password-hash 0.6` and re-exports it. Check if direct dep is needed or if `use argon2::password_hash::PasswordHash` works. Adjust import accordingly.

```rust
pub fn routes(pool: SqlitePool) -> Router {
    Router::new()
        .route("/register", post(register))
        .route("/login", post(login))
        .route("/logout", post(logout))
        .route("/2fa", post(verify_2fa))
        .with_state(pool)
}

#[derive(serde::Deserialize)]
struct RegisterRequest {
    email: String,
    password: String,
    name: Option<String>,
}

async fn register(
    State(pool): State<SqlitePool>,
    auth_session: AuthSession,
    Json(req): Json<RegisterRequest>,
) -> Result<Json<serde_json::Value>, AppError> {
    // Check registration enabled
    if !kartoteka_db::server_config::is_registration_enabled(&pool).await
        .map_err(|e| AppError::Internal(e.to_string()))? {
        return Err(AppError::Forbidden);
    }

    // Check email not taken
    if kartoteka_db::users::find_by_email(&pool, &req.email).await
        .map_err(|e| AppError::Internal(e.to_string()))?.is_some() {
        return Err(AppError::Validation("email_taken".into()));
    }

    // First user = admin
    let user_count = kartoteka_db::users::count(&pool).await
        .map_err(|e| AppError::Internal(e.to_string()))?;
    let role = if user_count == 0 { "admin" } else { "user" };

    // Create user
    let user = kartoteka_db::users::create(&pool, &req.email, req.name.as_deref(), role).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    // Hash password
    let salt = SaltString::generate(&mut OsRng);
    let hash = Argon2::default()
        .hash_password(req.password.as_bytes(), &salt)
        .map_err(|e| AppError::Internal(e.to_string()))?
        .to_string();

    // Store auth method
    kartoteka_db::auth_methods::create(&pool, &user.id, "password", &req.email, Some(&hash)).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    // Auto-login
    auth_session.login(&user).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    Ok(Json(serde_json::json!({
        "id": user.id,
        "email": user.email,
        "role": user.role,
    })))
}

async fn login(
    State(pool): State<SqlitePool>,
    auth_session: AuthSession,
    session: Session,
    Json(creds): Json<Credentials>,
) -> Result<Json<serde_json::Value>, AppError> {
    let user = auth_session.authenticate(creds).await
        .map_err(|e| AppError::Internal(e.to_string()))?
        .ok_or(AppError::Unauthorized)?;

    // Check if 2FA is enabled
    let totp = kartoteka_db::totp::find(&pool, &user.id).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    if totp.as_ref().is_some_and(|t| t.verified) {
        // Don't complete login yet — store pending user ID
        session.insert(PENDING_USER_ID_KEY, &user.id).await
            .map_err(|e| AppError::Internal(e.to_string()))?;
        return Ok(Json(serde_json::json!({
            "status": "2fa_required",
        })));
    }

    // No 2FA — complete login
    auth_session.login(&user).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    let return_to: Option<String> = session.get(RETURN_TO_KEY).await
        .unwrap_or(None);
    let _ = session.remove::<String>(RETURN_TO_KEY).await;

    Ok(Json(serde_json::json!({
        "status": "ok",
        "user": { "id": user.id, "email": user.email, "role": user.role },
        "return_to": return_to,
    })))
}

async fn verify_2fa(
    State(pool): State<SqlitePool>,
    auth_session: AuthSession,
    session: Session,
    Json(body): Json<TwoFaRequest>,
) -> Result<Json<serde_json::Value>, AppError> {
    let pending_user_id: String = session.get(PENDING_USER_ID_KEY).await
        .map_err(|e| AppError::Internal(e.to_string()))?
        .ok_or(AppError::Unauthorized)?;

    let totp_secret = kartoteka_db::totp::find(&pool, &pending_user_id).await
        .map_err(|e| AppError::Internal(e.to_string()))?
        .ok_or(AppError::Unauthorized)?;

    let totp = totp_rs::TOTP::new(
        totp_rs::Algorithm::SHA1,
        6, 1, 30,
        totp_rs::Secret::Encoded(totp_secret.secret).to_bytes().unwrap(),
    ).map_err(|e| AppError::Internal(e.to_string()))?;

    if !totp.check_current(&body.code).map_err(|e| AppError::Internal(e.to_string()))? {
        return Err(AppError::Validation("invalid_totp_code".into()));
    }

    // 2FA passed — complete login
    let _ = session.remove::<String>(PENDING_USER_ID_KEY).await;

    let user = kartoteka_db::users::find_by_id(&pool, &pending_user_id).await
        .map_err(|e| AppError::Internal(e.to_string()))?
        .ok_or(AppError::Unauthorized)?;

    auth_session.login(&user).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    let return_to: Option<String> = session.get(RETURN_TO_KEY).await
        .unwrap_or(None);
    let _ = session.remove::<String>(RETURN_TO_KEY).await;

    Ok(Json(serde_json::json!({
        "status": "ok",
        "user": { "id": user.id, "email": user.email, "role": user.role },
        "return_to": return_to,
    })))
}

#[derive(serde::Deserialize)]
struct TwoFaRequest {
    code: String,
}

    auth_session.logout().await
        .map_err(|e| AppError::Internal(e.to_string()))?;
    Ok(Json(serde_json::json!({ "status": "ok" })))
}
```

```rust
/// POST /auth/totp/setup — generate TOTP secret, return QR URI
pub async fn setup(
    State(pool): State<SqlitePool>,
    auth_session: AuthSession,
) -> Result<Json<serde_json::Value>, AppError> {
    let user = auth_session.user.ok_or(AppError::Unauthorized)?;

    let secret = Secret::generate_secret();
    let secret_base32 = secret.to_encoded().to_string();

    let totp = TOTP::new(
        Algorithm::SHA1, 6, 1, 30,
        secret.to_bytes().unwrap(),
    ).map_err(|e| AppError::Internal(e.to_string()))?;

    let otpauth_url = totp.get_url(&user.email, "Kartoteka");

    kartoteka_db::totp::upsert(&pool, &user.id, &secret_base32).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    Ok(Json(serde_json::json!({
        "secret": secret_base32,
        "otpauth_url": otpauth_url,
    })))
}

/// POST /auth/totp/verify — verify code, mark as verified
pub async fn verify(
    State(pool): State<SqlitePool>,
    auth_session: AuthSession,
    Json(body): Json<VerifyRequest>,
) -> Result<Json<serde_json::Value>, AppError> {
    let user = auth_session.user.ok_or(AppError::Unauthorized)?;

    let totp_secret = kartoteka_db::totp::find(&pool, &user.id).await
        .map_err(|e| AppError::Internal(e.to_string()))?
        .ok_or(AppError::NotFound("totp_not_setup".into()))?;

    let totp = TOTP::new(
        Algorithm::SHA1, 6, 1, 30,
        Secret::Encoded(totp_secret.secret).to_bytes().unwrap(),
    ).map_err(|e| AppError::Internal(e.to_string()))?;

    if !totp.check_current(&body.code).map_err(|e| AppError::Internal(e.to_string()))? {
        return Err(AppError::Validation("invalid_totp_code".into()));
    }

    kartoteka_db::totp::mark_verified(&pool, &user.id).await
        .map_err(|e| AppError::Internal(e.to_string()))?;

    Ok(Json(serde_json::json!({ "status": "verified" })))
}
```

```rust
use axum::{extract::Request, http::StatusCode, middleware::Next, response::Response, Json};

use super::backend::AuthSession;

pub async fn admin_required(
    auth_session: AuthSession,
    request: Request,
    next: Next,
) -> Result<Response, (StatusCode, Json<serde_json::Value>)> {
    let user = auth_session.user.ok_or_else(|| {
        (StatusCode::UNAUTHORIZED, Json(serde_json::json!({"error": "unauthorized"})))
    })?;

    if user.role != "admin" {
        return Err((StatusCode::FORBIDDEN, Json(serde_json::json!({"error": "admin_required"}))));
    }

    Ok(next.run(request).await)
}
```
